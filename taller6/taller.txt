Y = fft(S);
P2 = abs(Y/L);                % Normalizar
P1 = P2(1:L/2+1);             % Tomar la mitad del espectro
P1(2:end-1) = 2*P1(2:end-1);    % Multiplicar por 2 la mitad del espectro
f = Fs*(0:(L/2))/L;
plot(f,P1)
title('Single-Sided Amplitude Spectrum of S(t)')
xlabel('f (Hz)')
ylabel('|P1(f)|')
